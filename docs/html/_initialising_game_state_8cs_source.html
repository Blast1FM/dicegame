44 public override async Task Update()
45 {
46 // This will break if someone disconnects lol
47 if (_controller._players.Count == _controller.MaxPlayerCount)
48 {
49 await allStateChangeMessagesSent.Task;
50 _controller._stateManager.ChangeGameState(this, "pregame");
51 }
52 else
53 {
54 // Handle initialisation packets
55 foreach (var connection in _controller._unprocessedConnections)
56 {
57 if(!_initialisationTriesPerConnection.ContainsKey(connection))
58 {
59 _initialisationTriesPerConnection[connection] = 0;
60 }
61 _initialisationTriesPerConnection[connection]++;
62
63 if(_initialisationTriesPerConnection[connection] > _maxInitialisationTries)
64 {
65 await _controller.RefuseClientConnection(connection, $"Exceeded retry limit: {_maxInitialisationTries}");
66 continue;
67 }
68
69 Packet maybeInitPacket = await connection.ReceivePacket();
70 try
71 {
72 _requestRouter.RouteRequest(maybeInitPacket, connection);
73 }
74 catch (Exception e)
75 {
76 Console.WriteLine($"Server error when routing request from {connection.Socket.RemoteEndPoint}:{e.Message}");
77 await connection.SendErrorPacket(maybeInitPacket, $"Failed to route request: {e.Message}");
78 }
79 }

94 public async void HandleInitialiseRequest(Packet packet, HHTPClient clientConnection)
95 {
96 try
97 {
98 PlayerInfoMessage initMessage = JsonSerializer.Deserialize<PlayerInfoMessage>(packet.Payload) ?? throw new ArgumentNullException($"Player init message is null");
99
100 Player player = new(initMessage.PlayerInfo, clientConnection);
101
102 Packet okPacket = new(StatusCode.Ok,packet.Header.ProtocolMethod, packet.Header.ResourceIdentifier,"");
103
104 await clientConnection.SendPacket(okPacket);
105
106 _controller._players.Add(player);
107 _controller._unprocessedConnections.Remove(clientConnection);
108 }
109 catch (Exception e)
110 {
111 // Rollbacks shouldn't be neccessary
112 await clientConnection.SendErrorPacket(packet, $"Server error, could not initialise: {e.Message}");
113 }
114 }

116 public async void HandleGetConnectedPlayersRequest(Packet packet, HHTPClient clientConnection)
117 {
118 try
119 {
120 ConnectedPlayerListMessage connectedPlayersDTO = new(_controller._players.Select(p => p.PlayerInfo).ToList());
121 string message = JsonSerializer.Serialize(connectedPlayersDTO);
122 Packet response = new(StatusCode.Ok, packet.Header.ProtocolMethod, packet.Header.ResourceIdentifier, message);
123 await clientConnection.SendPacket(response);
124 }
125 catch (JsonException e)
126 {
127 Packet errorPacket = new(StatusCode.Error, packet.Header.ProtocolMethod, packet.Header.ResourceIdentifier, $"Serialisation error: {e.Message}");
128 }
129 catch (Exception e)
130 {
131 Packet errorPacket = new(StatusCode.Error, packet.Header.ProtocolMethod, packet.Header.ResourceIdentifier, $"Unhandled server error:{e.Message}");
132 }
133 }

135 public async void HandleGetStateStatusRequest(Packet packet, HHTPClient clientConnection)

142 // Ideally retry logic but cba

147 }
148 } else

150 GameStateStatusMessage message = new("initialising", "initialising");
